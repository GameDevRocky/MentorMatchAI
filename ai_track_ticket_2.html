1. Weighted Field Matching

Implement field weighting so that expertise areas are more important than names:

create_weighted_mentor_text <- function(mentor_data) {
  weighted_texts <- c()
  for (i in 1:nrow(mentor_data)) {
    mentor <- mentor_data[i, ]
    
    # Expertise is most important (40% weight = repeat 4 times)
    expertise_text <- paste(rep(mentor$expertise, 4), collapse = " ")
    
    # Bio is second most important (30% weight = repeat 3 times) 
    bio_text <- paste(rep(mentor$bio, 3), collapse = " ")
    
    # Title is third (20% weight = repeat 2 times)
    title_text <- paste(rep(mentor$title, 2), collapse = " ")
    
    # Combine everything
    combined_text <- paste(mentor$name, expertise_text, bio_text, title_text)
    weighted_texts <- c(weighted_texts, combined_text)
  }
  return(weighted_texts)
}
